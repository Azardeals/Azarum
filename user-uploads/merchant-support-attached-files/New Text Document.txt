SELECT SQL_CALC_FOUND_ROWS c.company_id, d.deal_id, od.*, o.*, SUM((od.od_qty+od.od_gift_qty)*od.od_deal_price) as totalAmount FROM `tbl_companies` c INNER JOIN `tbl_deals` d on c.company_id=d.deal_company INNER JOIN `tbl_order_deals` od on od.od_deal_id=d.deal_id INNER JOIN `tbl_orders` o on o.order_id=od.od_order_id WHERE c.`company_rep_id` = '1' AND o.`order_payment_status` != '0' GROUP BY totalAmount HAVING `totalAmount` > '1500' limit 0, 2000

SELECT SQL_CALC_FOUND_ROWS od.*, o.*, SUM((od.od_qty+od.od_gift_qty)*od.od_deal_price) as totalAmount FROM `tbl_orders` o INNER JOIN `tbl_order_deals` od on o.order_id=od.od_order_id INNER JOIN `tbl_deals` d on od.od_deal_id=d.deal_id WHERE o.`order_payment_status` != '0' AND od.`od_deal_id` IN ('7', '8', '9', '10', '11', '13', '15', '16', '22', '23', '25', '26', '27', '30', '31', '32', '33', '34', '36', '37', '38', '42', '43', '44', '46', '52', '53', '74', '92', '93', '94', '109') limit 0, 2000
